function isArmstrong($num){ 
$sum = 0; 
$temp = $num; 
$num_digits = strlen($num); 
while ($temp!= 0){ 
$digit = $temp % 10; 
$sum += pow($digit, $num_digits); 
$temp = intval($temp / 10); 
} 
return $sum == $num; 
}
